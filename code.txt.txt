Console.WriteLine("Игнатьев В.Д. РПИб-о25");
// Читаем строку ввода от пользователя
string input = Console.ReadLine();

// Проверяем, что ввод не пустой
if (string.IsNullOrEmpty(input))
{
    Console.WriteLine("Error: Input cannot be empty.");
    return;
}

// Удаляем начальные и конечные пробелы из строки
input = input.Trim();

// Пытаемся преобразовать строку в целое число
// Если не удалось — выводим ошибку
if (!int.TryParse(input, out int n))
{
    Console.WriteLine("Error: Please enter a valid integer.");
    return;
}

// Проверяем, делится ли число на 2 или на 5
// Если да — число из единиц, состоящее только из 1, не может делиться на n нацело
// (так как число из единиц не делится на 2 и на 5)
if (n % 2 == 0 || n % 5 == 0)
{
    Console.WriteLine("NO");
    return;
}

// Массив для отслеживания уже встреченных остатков
// Используется для обнаружения цикла (если остаток повторился — цикл бесконечный)
bool[] seen = new bool[n];

// Вычисляем начальный остаток: 1 % n (остаток от деления числа "1" на n)
int remainder = 1 % n;

// Счётчик количества единиц в искомом числе
int count = 1;

// Цикл: ищем, пока остаток не станет равным 0
// (остаток 0 означает, что число из count единиц делится на n нацело)
while (remainder != 0)
{
    // Если такой остаток уже встречался — мы зациклились
    // Значит, число из единиц, делящееся на n, не существует
    if (seen[remainder])
    {
        Console.WriteLine("NO");
        return;
    }

    // Отмечаем текущий остаток как встреченный
    seen[remainder] = true;

    // Вычисляем следующий остаток по формуле:
    // (остаток * 10 + 1) % n
    // Это эквивалентно добавлению ещё одной цифры '1' в конец числа
    // Например: 1 → 11 → 111 → 1111 и т.д.
    remainder = (remainder * 10 + 1) % n;

    // Увеличиваем счётчик единиц
    count++;
}

// Создаём строку из count символов '1' и выводим её
Console.WriteLine(new string('1', count));
